## 回溯算法

```js
let result = []
function backtrack(路径, 选择列表){
  // 结束条件
  if(满足结束条件){
    result.push(路径)
    return
  }
  // 进行选择
  for(let 选择 of 选择列表){
    // 做选择加入路径
    路径.push(选择)
    // 进入下一层
    backtrack(路径, 选择列表)
    // 撤销选择，进行for循环的下一轮
    路径.pop(选择)
  } 
}
```
